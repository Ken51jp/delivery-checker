function check() { const input = document.getElementById('addressInput').value.trim(); const resultDiv = document.getElementById('result'); const debug = document.getElementById('debug-info'); if (!input) return; // 【新機能】入力された文字をスペース（全角・半角両方）で分割して配列にする const keywords = input.split(/[\s　]+/).filter(k => k !== ""); const found = deliveryData.find(item => { if (!item.full) return false; // すべてのキーワードがCSVの住所（県+市+区）に含まれているかチェック // 例：「北海道　清田」→「北海道札幌市清田区」の中に両方あればヒット return keywords.every(keyword => item.full.includes(keyword)); }); if (!found) { resultDiv.innerHTML = "該当データなし"; resultDiv.className = ""; return; } let message = ""; let className = ""; if (found.type === "1") { message = "✨ 明日の午前中に届きます"; className = "type1"; } else if (found.type === "2") { message = "🚚 明日の午後以降に届きます"; className = "type2"; } else if (found.type === "3") { message = "⏳ 翌々日の到着予定です"; className = "type3"; } else { message = `区分不明 (${found.type})`; className = ""; } resultDiv.innerHTML = message + `
(判定エリア: ${found.full})`; resultDiv.className = className; }
